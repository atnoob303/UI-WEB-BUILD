function selectEl(id) {
  selected = id;
  elements.forEach(el => {
    const d = document.getElementById(el.id);
    if (d) {
      d.classList.toggle('selected-el', el.id === id);
      if (el.id !== id && !(el.modifiers||{}).UIStroke) d.style.outline = 'none';
    }
  });
  renderProps();
  renderElementsList();
}

function startDrag(el, e) {
  saveHistory();
  const offX = e.clientX - el.x, offY = e.clientY - el.y;
  const mm = ev => { el.x = Math.max(0, ev.clientX - offX); el.y = Math.max(0, ev.clientY - offY); renderEl(el); updateCanvasInfo(el); };
  const mu = () => { document.removeEventListener('mousemove', mm); document.removeEventListener('mouseup', mu); };
  document.addEventListener('mousemove', mm);
  document.addEventListener('mouseup', mu);
}

function startResize(el, e) {
  saveHistory();
  const sx = e.clientX, sy = e.clientY, sw = el.w, sh = el.h;
  const mm = ev => {
    el.w = Math.max(20, sw + (ev.clientX - sx));
    el.h = Math.max(20, sh + (ev.clientY - sy));
    renderEl(el); renderProps(); updateCanvasInfo(el);
  };
  const mu = () => { document.removeEventListener('mousemove', mm); document.removeEventListener('mouseup', mu); };
  document.addEventListener('mousemove', mm);
  document.addEventListener('mouseup', mu);
}

function updateCanvasInfo(el) {
  document.getElementById('canvas-info').textContent = `${el.type} • ${Math.round(el.x)},${Math.round(el.y)} • ${Math.round(el.w)}×${Math.round(el.h)}`;
}

  const mm = ev => {

    el.x = Math.min(drawStart.x, nx); el.y = Math.min(drawStart.y, ny);

    if (el.h < 20) el.h = defaultHeight(type);

    document.removeEventListener('mousemove', mm);
    document.removeEventListener('mouseup', mu);
  };
  document.addEventListener('mousemove', mm);
  document.addEventListener('mouseup', mu);
};

function addModifierToSelected(modType) {
  const el = elements.find(e => e.id === selected);
  if (!el) { showToast('⚠ Chọn element trước!'); return; }
  if (!el.modifiers) el.modifiers = {};
  if (el.modifiers[modType]) { showToast(`${modType} already added!`); return; }
  saveHistory();
  el.modifiers[modType] = JSON.parse(JSON.stringify(MODIFIER_DEFAULTS[modType] || {}));
  renderEl(el);
  renderProps();
  showToast(`✓ Added ${modType}`);
}

function removeModifier(modType) {
  const el = elements.find(e => e.id === selected);
  if (!el || !el.modifiers) return;
  saveHistory();
  delete el.modifiers[modType];
  renderEl(el);
  renderProps();
}

  renderEl(el);
}

function duplicateSelected() {
  if (!selected) return;
  const orig = elements.find(e => e.id === selected);
  if (!orig) return;
  saveHistory();
  const copy = JSON.parse(JSON.stringify(orig));
  copy.id = 'el_' + (++idCounter);
  copy.name = orig.type + idCounter;
  copy.x += 20; copy.y += 20;
  elements.push(copy);
  renderEl(copy);
  selectEl(copy.id);
  renderElementsList();
}

function deleteSelected() {
  if (!selected) return;
  saveHistory();
  const d = document.getElementById(selected);
  if (d) d.remove();
  elements = elements.filter(e => e.id !== selected);
  selected = null;
  renderProps(); renderElementsList(); syncHint();
}

function clearAll() {
  if (elements.length && !confirm('Xóa tất cả elements?')) return;
  saveHistory();
  elements.forEach(el => { const d = document.getElementById(el.id); if (d) d.remove(); });
  elements = []; selected = null;
  renderProps(); renderElementsList(); syncHint();
}

function syncHint() {
  document.getElementById('empty-hint').style.display = elements.length ? 'none' : 'flex';
  document.getElementById('el-empty').style.display = elements.length ? 'none' : 'block';
}

function section(title, content) {
  return `<div class="panel-section"><div class="section-header">${title}</div><div class="section-body">${content}</div></div>`;
}

function numRow(id, label, key, min, max, val, step=1) {
  const isMod = key.startsWith('$');
  const onChange = isMod
    ? `modSetPath('${id}','${key.slice(1)}',+this.value)`
    : `propSet('${id}','${key}',+this.value)`;
  const lblId = 'lbl_' + key.replace(/\./g,'_') + '_' + id;
  return `<div class="prop-row">
    <span class="prop-label">${label}</span>
    <input type="range" class="prop-input" min="${min}" max="${max}" step="${step}" value="${val}"
      style="flex:1" oninput="${onChange};document.getElementById('${lblId}').textContent=parseFloat(this.value).toFixed(step<1?2:0)"/>
    <span class="prop-val" id="${lblId}">${typeof val==='number'?val.toFixed(step<1?2:0):val}</span>
  </div>`;
}

function colRow(id, label, key, val) {
  const hexVal = val ? rgbToHex(val) : '#313244';
  const isMod = key.startsWith('$');
  const onChange = isMod
    ? `modSetPath('${id}','${key.slice(1)}',hexToRgb(this.value))`
    : `propSetRgb('${id}','${key}',this.value)`;
  return `<div class="prop-row">
    <span class="prop-label">${label}</span>
    <input type="color" class="prop-input" value="${hexVal}" oninput="${onChange}"/>
  </div>`;
}

function txtRow(id, label, key, val) {
  return `<div class="prop-row">
    <span class="prop-label">${label}</span>
    <input type="text" class="prop-input" value="${(val||'').replace(/"/g,'&quot;')}"
      oninput="propSet('${id}','${key}',this.value)"/>
  </div>`;
}

function chkRow(id, label, key, val) {
  const isMod = key.startsWith('$');
  const onChange = isMod
    ? `modSetPath('${id}','${key.slice(1)}',this.checked)`
    : `propSet('${id}','${key}',this.checked)`;
  return `<div class="prop-row">
    <span class="prop-label">${label}</span>
    <input type="checkbox" ${val?'checked':''} onchange="${onChange};renderEl(elements.find(e=>e.id==='${id}'));renderProps()"/>
  </div>`;
}

// ─────────────────────────────────────────────
// PROP SETTERS
// ─────────────────────────────────────────────
function propSet(id, key, val) {
  const el = elements.find(e => e.id === id);
  if (!el) return;
  el[key] = val;
  renderEl(el);
}

function propSetRgb(id, key, hex) {
  const el = elements.find(e => e.id === id);
  if (!el) return;
  el[key] = hexToRgb(hex);
  el.bg = hex; // sync display
  renderEl(el);
}

function modSet(id, modType, key, val) {
  const el = elements.find(e => e.id === id);
  if (!el || !el.modifiers || !el.modifiers[modType]) return;
  el.modifiers[modType][key] = val;
  renderEl(el);
}

function modSetPath(id, path, val) {
  const el = elements.find(e => e.id === id);
  if (!el) return;
  const parts = path.split('.');
  const modType = parts[0], key = parts[1];
  if (!el.modifiers || !el.modifiers[modType]) return;
  el.modifiers[modType][key] = val;
  renderEl(el);
}

// ─────────────────────────────────────────────
// COLOR UTILS
// ─────────────────────────────────────────────
function hexToRgb(hex) {
  const r = parseInt(hex.slice(1,3),16), g = parseInt(hex.slice(3,5),16), b = parseInt(hex.slice(5,7),16);
  return {r,g,b};
}

function rgbToHex(c) {
  if (!c) return '#313244';
  const r = Math.round(c.r||0), g = Math.round(c.g||0), b = Math.round(c.b||0);
  return '#' + [r,g,b].map(x=>x.toString(16).padStart(2,'0')).join('');
}

    out += `-- ── ${el.type}: ${el.name} ──\n`;
    out += `local ${vName} = Instance.new("${el.type}")\n`;

    // Parent
    if (el.type === 'ScreenGui') {
      out += `${vName}.Parent = playerGui\n`;
    } else if (i > 0) {
      const parentEl = elements[0];
      const parentName = (parentEl.name||parentEl.type+'0').replace(/[^a-zA-Z0-9_]/g,'_');

    } else {
      out += `${vName}.Parent = playerGui\n`;
    }

    if (el.type !== 'ScreenGui') {
      out += `${vName}.Position = UDim2.new(${el.posScaleX||0}, ${Math.round(el.x)}, ${el.posScaleY||0}, ${Math.round(el.y)})\n`;
      out += `${vName}.Size = UDim2.new(${el.sizeScaleX||0}, ${Math.round(el.w)}, ${el.sizeScaleY||0}, ${Math.round(el.h)})\n`;
      out += `${vName}.AnchorPoint = Vector2.new(${el.anchorX||0}, ${el.anchorY||0})\n`;
      out += `${vName}.ZIndex = ${el.zIndex||0}\n`;
      if (el.bgColor) out += `${vName}.BackgroundColor3 = Color3.fromRGB(${Math.round(el.bgColor.r||0)}, ${Math.round(el.bgColor.g||0)}, ${Math.round(el.bgColor.b||0)})\n`;
      const transparency = 1 - (typeof el.opacity === 'number' ? el.opacity : 1);
      out += `${vName}.BackgroundTransparency = ${transparency.toFixed(2)}\n`;
      if (el.visible === false) out += `${vName}.Visible = false\n`;
    }

    // Type specific
    if ((el.type === 'TextLabel' || el.type === 'TextButton') && el.text) {
      out += `${vName}.Text = "${(el.text||'').replace(/"/g,'\\"')}"\n`;
      if (el.textColor) out += `${vName}.TextColor3 = Color3.fromRGB(${Math.round(el.textColor.r||0)}, ${Math.round(el.textColor.g||0)}, ${Math.round(el.textColor.b||0)})\n`;
      out += `${vName}.TextSize = ${el.textSize||14}\n`;
      out += `${vName}.Font = Enum.Font.${el.font||'GothamMedium'}\n`;
      out += `${vName}.TextXAlignment = Enum.TextXAlignment.${el.textXAlignment||'Left'}\n`;
      out += `${vName}.TextYAlignment = Enum.TextYAlignment.${el.textYAlignment||'Center'}\n`;
      if (el.textWrapped) out += `${vName}.TextWrapped = true\n`;
      if (el.textScaled) out += `${vName}.TextScaled = true\n`;
      if (el.richText) out += `${vName}.RichText = true\n`;
    }

    if ((el.type === 'ImageLabel' || el.type === 'ImageButton') && el.image) {
      out += `${vName}.Image = "${el.image}"\n`;
      if (el.imageColor) out += `${vName}.ImageColor3 = Color3.fromRGB(${Math.round(el.imageColor.r||0)}, ${Math.round(el.imageColor.g||0)}, ${Math.round(el.imageColor.b||0)})\n`;
      out += `${vName}.ImageTransparency = ${el.imageTransparency||0}\n`;
      out += `${vName}.ScaleType = Enum.ScaleType.${el.scaleType||'Stretch'}\n`;
    }

    if (el.type === 'ScrollingFrame') {
      out += `${vName}.ScrollBarThickness = ${el.scrollBarThickness||6}\n`;
      if (el.scrollBarColor) out += `${vName}.ScrollBarImageColor3 = Color3.fromRGB(${Math.round(el.scrollBarColor.r||0)}, ${Math.round(el.scrollBarColor.g||0)}, ${Math.round(el.scrollBarColor.b||0)})\n`;
      out += `${vName}.CanvasSize = UDim2.new(0, 0, 0, ${el.canvasSizeY||200})\n`;
      if (el.scrollEnabled === false) out += `${vName}.ScrollingEnabled = false\n`;
    }

    if (el.type === 'VideoFrame') {
      out += `${vName}.Video = "${el.videoId||'rbxassetid://0'}"\n`;
      out += `${vName}.Volume = ${el.volume||0.5}\n`;
      if (el.looped) out += `${vName}.Looped = true\n`;
      if (el.playing) out += `${vName}:Play()\n`;
    }

    if (el.type === 'ScreenGui') {
      if (el.enabled === false) out += `${vName}.Enabled = false\n`;
      out += `${vName}.DisplayOrder = ${el.displayOrder||0}\n`;
      if (el.resetOnSpawn === false) out += `${vName}.ResetOnSpawn = false\n`;
      if (el.ignoreGuiInset) out += `${vName}.IgnoreGuiInset = true\n`;
    }

    if (el.type === 'CanvasGroup') {
      out += `${vName}.GroupTransparency = ${el.groupTransparency||0}\n`;
    }

    if (el.type === 'TextButton' || el.type === 'ImageButton') {
      if (el.autoButtonColor === false) out += `${vName}.AutoButtonColor = false\n`;
      if (el.modal) out += `${vName}.Modal = true\n`;
    }

    // Modifiers
    const mods = el.modifiers || {};
    Object.keys(mods).forEach(mk => {
      const md = mods[mk];
      const mvName = mk + '_' + vName;
      out += `\nlocal ${mvName} = Instance.new("${mk}")\n`;
      out += `${mvName}.Parent = ${vName}\n`;

      if (mk === 'UICorner') out += `${mvName}.CornerRadius = UDim.new(0, ${md.cornerRadius||0})\n`;
      if (mk === 'UIGradient') {
        const c0 = hexToRgb(md.color0||'#7c6af7'), c1 = hexToRgb(md.color1||'#22d3ee');
        out += `${mvName}.Color = ColorSequence.new({ColorSequenceKeypoint.new(0, Color3.fromRGB(${c0.r},${c0.g},${c0.b})), ColorSequenceKeypoint.new(1, Color3.fromRGB(${c1.r},${c1.g},${c1.b}))})\n`;
        out += `${mvName}.Rotation = ${md.rotation||0}\n`;
        if (md.enabled === false) out += `${mvName}.Enabled = false\n`;
      }
      if (mk === 'UIStroke') {
        const sc = hexToRgb(md.color||'#7c6af7');
        out += `${mvName}.Color = Color3.fromRGB(${sc.r},${sc.g},${sc.b})\n`;
        out += `${mvName}.Thickness = ${md.thickness||2}\n`;
        out += `${mvName}.Transparency = ${md.transparency||0}\n`;
        out += `${mvName}.ApplyStrokeMode = Enum.ApplyStrokeMode.${md.applyStrokeMode||'Border'}\n`;
        if (md.enabled === false) out += `${mvName}.Enabled = false\n`;
      }
      if (mk === 'UIPadding') {
        out += `${mvName}.PaddingTop = UDim.new(0, ${md.top||0})\n`;
        out += `${mvName}.PaddingBottom = UDim.new(0, ${md.bottom||0})\n`;
        out += `${mvName}.PaddingLeft = UDim.new(0, ${md.left||0})\n`;
        out += `${mvName}.PaddingRight = UDim.new(0, ${md.right||0})\n`;
      }
      if (mk === 'UIScale') out += `${mvName}.Scale = ${md.scale||1}\n`;
      if (mk === 'UIAspectRatioConstraint') {
        out += `${mvName}.AspectRatio = ${md.aspectRatio||1}\n`;
        out += `${mvName}.AspectType = Enum.AspectType.${md.aspectType||'FitWithinMaxSize'}\n`;
        out += `${mvName}.DominantAxis = Enum.DominantAxis.${md.dominantAxis||'Width'}\n`;
      }
      if (mk === 'UISizeConstraint') {
        out += `${mvName}.MinSize = Vector2.new(${md.minSizeX||0}, ${md.minSizeY||0})\n`;
        out += `${mvName}.MaxSize = Vector2.new(${md.maxSizeX||999}, ${md.maxSizeY||999})\n`;
      }
      if (mk === 'UITextSizeConstraint') {
        out += `${mvName}.MinTextSize = ${md.minTextSize||6}\n`;
        out += `${mvName}.MaxTextSize = ${md.maxTextSize||100}\n`;
      }
      if (mk === 'UIListLayout') {
        out += `${mvName}.FillDirection = Enum.FillDirection.${md.fillDirection||'Vertical'}\n`;
        out += `${mvName}.HorizontalAlignment = Enum.HorizontalAlignment.${md.horizontalAlignment||'Left'}\n`;
        out += `${mvName}.VerticalAlignment = Enum.VerticalAlignment.${md.verticalAlignment||'Top'}\n`;
        out += `${mvName}.SortOrder = Enum.SortOrder.${md.sortOrder||'LayoutOrder'}\n`;
        out += `${mvName}.Padding = UDim.new(0, ${md.padding||0})\n`;
        if (md.wraps) out += `${mvName}.Wraps = true\n`;
      }
      if (mk === 'UIGridLayout') {
        out += `${mvName}.CellSize = UDim2.new(0, ${md.cellSize||100}, 0, ${md.cellSize||100})\n`;
        out += `${mvName}.CellPaddingX = UDim.new(0, ${md.cellPaddingX||4})\n`;
        out += `${mvName}.CellPadding = UDim2.new(0, ${md.cellPaddingX||4}, 0, ${md.cellPaddingY||4})\n`;
        out += `${mvName}.FillDirection = Enum.FillDirection.${md.fillDirection||'Horizontal'}\n`;
        out += `${mvName}.HorizontalAlignment = Enum.HorizontalAlignment.${md.horizontalAlignment||'Left'}\n`;
        out += `${mvName}.SortOrder = Enum.SortOrder.${md.sortOrder||'LayoutOrder'}\n`;
        out += `${mvName}.StartCorner = Enum.StartCorner.${md.startCorner||'TopLeft'}\n`;
      }
      if (mk === 'UITableLayout') {
        out += `${mvName}.FillEmptySpaceColumns = ${md.fillEmptySpaceColumns?'true':'false'}\n`;
        out += `${mvName}.FillEmptySpaceRows = ${md.fillEmptySpaceRows?'true':'false'}\n`;
        out += `${mvName}.Padding = UDim2.new(0, ${md.padding||0}, 0, ${md.padding||0})\n`;
      }
      if (mk === 'UIPageLayout') {
        out += `${mvName}.Animated = ${md.animated!==false?'true':'false'}\n`;
        out += `${mvName}.AnimationDirection = Enum.AnimationDirection.${md.animationDirection||'Horizontal'}\n`;
        out += `${mvName}.Circular = ${md.circular?'true':'false'}\n`;
        out += `${mvName}.EasingStyle = Enum.EasingStyle.${md.easingStyle||'Quad'}\n`;
        out += `${mvName}.EasingDirection = Enum.EasingDirection.${md.easingDirection||'Out'}\n`;
        out += `${mvName}.Padding = UDim.new(0, ${md.padding||0})\n`;
      }
      if (mk === 'UIFlexItem') {
        out += `${mvName}.FlexMode = Enum.UIFlexMode.${md.flexMode||'Fill'}\n`;
        out += `${mvName}.GrowRatio = ${md.growRatio||1}\n`;
        out += `${mvName}.ShrinkRatio = ${md.shrinkRatio||1}\n`;
        out += `${mvName}.ItemLineAlignment = Enum.ItemLineAlignment.${md.itemLineAlignment||'Automatic'}\n`;
      }
    });

    // Click handler for buttons
    if (el.type === 'TextButton' || el.type === 'ImageButton') {
      out += `\n${vName}.MouseButton1Click:Connect(function()\n\t-- TODO: add your logic here\nend)\n`;
    }

    out += '\n';
  });

  return out;
}

function buildHTMLCode() {
  let out = `<!DOCTYPE html>\n<html>\n<head>\n<meta charset="UTF-8">\n<title>Roblox UI Preview</title>\n<style>\nbody{margin:0;background:#0d0d14;font-family:Gotham,sans-serif}\n.ui-root{position:relative;width:800px;height:600px;overflow:hidden}\n</style>\n</head>\n<body>\n<div class="ui-root">\n`;
  elements.forEach(el => {
    if (el.type === 'ScreenGui') return;
    const bg = el.bgColor ? `rgb(${Math.round(el.bgColor.r)},${Math.round(el.bgColor.g)},${Math.round(el.bgColor.b)})` : '#222';
    const mods = el.modifiers || {};
    let style = `position:absolute;left:${el.x}px;top:${el.y}px;width:${el.w}px;height:${el.h}px;`;
    style += `opacity:${el.opacity||1};z-index:${el.zIndex||0};`;
    if (mods.UICorner) style += `border-radius:${mods.UICorner.cornerRadius||0}px;`;
    if (mods.UIGradient && mods.UIGradient.enabled!==false) style += `background:linear-gradient(${mods.UIGradient.rotation||0}deg,${mods.UIGradient.color0||'#7c6af7'},${mods.UIGradient.color1||'#22d3ee'});`;
    else style += `background:${bg};`;
    if (mods.UIStroke && mods.UIStroke.enabled!==false) style += `outline:${mods.UIStroke.thickness||2}px solid ${mods.UIStroke.color||'#7c6af7'};`;
    if (mods.UIPadding) style += `padding:${mods.UIPadding.top||0}px ${mods.UIPadding.right||0}px ${mods.UIPadding.bottom||0}px ${mods.UIPadding.left||0}px;`;
    if (el.type === 'TextLabel' || el.type === 'TextButton') {
      const tc = el.textColor ? `rgb(${Math.round(el.textColor.r)},${Math.round(el.textColor.g)},${Math.round(el.textColor.b)})` : '#fff';
      style += `display:flex;align-items:center;justify-content:${el.textXAlignment==='Center'?'center':el.textXAlignment==='Right'?'flex-end':'flex-start'};color:${tc};font-size:${el.textSize||14}px;font-weight:${el.type==='TextButton'?700:400};`;
      if (el.type === 'TextButton') style += `cursor:pointer;`;
      out += `  <div style="${style}">${el.text||''}</div>\n`;
    } else {
      out += `  <div style="${style}"></div>\n`;
    }
  });
  out += `</div>\n</body>\n</html>`;
  return out;
}

function showExport() {
  const code = exportTab === 'lua' ? buildLuaCode() : buildHTMLCode();
  document.getElementById('export-code').textContent = code;
  document.getElementById('export-stats').textContent = `${elements.length} elements • ${code.split('\n').length} lines`;
  document.getElementById('export-modal').classList.add('show');
}
function hideExport() { document.getElementById('export-modal').classList.remove('show'); }

function copyCode() {
  const code = exportTab === 'lua' ? buildLuaCode() : buildHTMLCode();
  navigator.clipboard.writeText(code).then(() => showToast('✅ Copied!'));
}

function downloadCode() {
  const code = exportTab === 'lua' ? buildLuaCode() : buildHTMLCode();
  const ext = exportTab === 'lua' ? '.lua' : '.html';
  const blob = new Blob([code], { type: 'text/plain' });
  const a = document.createElement('a'); a.href = URL.createObjectURL(blob);
  a.download = 'roblox-ui' + ext; a.click();
}

document.addEventListener('keydown', e => {
  const tag = document.activeElement.tagName;
  if (tag === 'INPUT' || tag === 'TEXTAREA' || tag === 'SELECT') return;
  if (e.key === 'Delete' || e.key === 'Backspace') deleteSelected();
  if (e.ctrlKey && e.key === 'z') { e.preventDefault(); undoAction(); }
  if (e.ctrlKey && e.key === 'd') { e.preventDefault(); duplicateSelected(); }
  if (e.key === 'Escape') { setTool('select'); selectEl(null); renderProps(); drawTool = null; }
  if (e.key === 'ArrowUp' && selected) { const el = elements.find(e=>e.id===selected); if(el){saveHistory();el.y=Math.max(0,el.y-(e.shiftKey?10:1));renderEl(el);} }
  if (e.key === 'ArrowDown' && selected) { const el = elements.find(e=>e.id===selected); if(el){saveHistory();el.y+=e.shiftKey?10:1;renderEl(el);} }
  if (e.key === 'ArrowLeft' && selected) { const el = elements.find(e=>e.id===selected); if(el){saveHistory();el.x=Math.max(0,el.x-(e.shiftKey?10:1));renderEl(el);} }
  if (e.key === 'ArrowRight' && selected) { const el = elements.find(e=>e.id===selected); if(el){saveHistory();el.x+=e.shiftKey?10:1;renderEl(el);} }
});
